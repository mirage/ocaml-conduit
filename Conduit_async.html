Connection establishment using the Async library
type 'a io = 'a Async.Std.Deferred.t 
type ic = Async.Std.Reader.t 
type oc = Async.Std.Writer.t 
type addr = [ `OpenSSL of string * Ipaddr.t * int
       | `TCP of Ipaddr.t * int
       | `Unix_domain_socket of string ] 
val connect : ?interrupt:unit io ->
       addr -> (ic * oc) io
type server = [ `OpenSSL of [ `Crt_file_path of string ] * [ `Key_file_path of string ]
       | `TCP ] 
val serve : ?max_connections:int ->
       ?max_pending_connections:int ->
       ?buffer_age_limit:Async.Std.Writer.buffer_age_limit ->
       ?on_handler_error:[ `Call of
                             ([< Async_extra.Import.Socket.Address.t ] as 'a) ->
                             exn -> unit
                         | `Ignore
                         | `Raise ] ->
       server ->
       ('a, 'b) Async.Std.Tcp.Where_to_listen.t ->
       ('a -> ic -> oc -> unit io) ->
       ('a, 'b) Async.Std.Tcp.Server.t io
