Module Conduit_async_ssl

module Conduit_async_ssl: sig .. end
TLS/SSL connection establishment using OpenSSL and Async
val ssl_connect : Async.Std.Reader.t ->
       Async.Std.Writer.t ->
       (Async.Std.Reader.t * Async.Std.Writer.t) Async.Std.Deferred.t
ssl_connect rd wr will establish a client TLS/SSL session over an existing pair of a rd Reader.t and wd Writer.t Async connections.
val ssl_listen : crt_file:string ->
       key_file:string ->
       Async.Std.Reader.t ->
       Async.Std.Writer.t ->
       (Async.Std.Reader.t * Async.Std.Writer.t) Async.Std.Deferred.t
ssl_listen ~crt_file ~key_file rd wr will establish a server TLS/SSL session over an existing pair of rd Reader.t and wd Writer.t Async connections.
